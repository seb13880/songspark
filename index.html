 const API_KEY = 'atk_28df805fe43e62aa5731f1abcd6ef3dd1910af5a';


 // Pills selection

 function setupPills(groupId, multi = false) {

   const group = document.getElementById(groupId);

   group.querySelectorAll('.pill').forEach(pill => {

     pill.addEventListener('click', () => {

       if (!multi) group.querySelectorAll('.pill').forEach(p => p.classList.remove('active'));

       pill.classList.toggle('active');

     });

   });

 }

 setupPills('style-pills', false);

 setupPills('mood-pills', false);

 setupPills('lang-pills', false);


 function getSelected(groupId) {

   const active = document.querySelectorAll(`#${groupId} .pill.active`);

   return Array.from(active).map(p => p.dataset.value).join(', ');

 }


 let progressInterval;


 function animateProgress(targetPct, duration = 3000) {

   clearInterval(progressInterval);

   const fill = document.getElementById('progress');

   let current = parseFloat(fill.style.width) || 0;

   const step = (targetPct - current) / (duration / 100);

   progressInterval = setInterval(() => {

     current += step;

     if (current >= targetPct) { current = targetPct; clearInterval(progressInterval); }

     fill.style.width = current + '%';

   }, 100);

 }


 function setStep(text, pct) {

   document.getElementById('loading-step').textContent = text;

   animateProgress(pct);

 }


 async function generate() {

   const theme = document.getElementById('theme').value.trim();

   if (!theme) {

     document.getElementById('theme').focus();

     return;

   }


   const style = getSelected('style-pills') || 'pop';

   const mood = getSelected('mood-pills') || 'mélancolique';

   const lang = getSelected('lang-pills') || 'French';


   const prompt = `${mood} ${style} song in ${lang} about: ${theme}`;

   const tags = `${style}, ${mood}, vocals, catchy, ${lang}`;


   // UI state

   document.getElementById('btn-gen').disabled = true;

   document.getElementById('loading').style.display = 'flex';

   document.getElementById('result').style.display = 'none';

   document.getElementById('error').style.display = 'none';

   document.getElementById('progress').style.width = '0%';


   setStep('Écriture des paroles…', 20);


   try {

     // Step 1: Generate

     const genRes = await fetch('https://api.apiframe.pro/music-generate', {

       method: 'POST',

       headers: {

         'Content-Type': 'application/json',

         'Authorization': API_KEY

       },

       body: JSON.stringify({

         prompt: prompt,

         tags: tags,

         title: `Ma chanson – ${style}`,

         make_instrumental: false,

         model: 'V3_5'

       })

     });


     const genData = await genRes.json();


     if (!genRes.ok || genData.error) {

       throw new Error(genData.error || genData.message || 'Erreur lors de la génération.');

     }


     const taskId = genData.task_id;

     if (!taskId) throw new Error('Réponse inattendue du serveur.');


     setStep('Composition musicale…', 45);


     // Step 2: Poll

     let audio_url = null, title = null, lyric = null;

     let attempts = 0;

     const maxAttempts = 60;


     while (attempts < maxAttempts) {

       await new Promise(r => setTimeout(r, 5000));

       attempts++;


       const fetchRes = await fetch(`https://api.apiframe.pro/music-fetch`, {

         method: 'POST',

         headers: {

           'Content-Type': 'application/json',

           'Authorization': API_KEY

         },

         body: JSON.stringify({ task_id: taskId })

       });


       const fetchData = await fetchRes.json();


       if (fetchData.status === 'failed') throw new Error('La génération a échoué. Réessaie.');


       if (fetchData.status === 'finished' || fetchData.status === 'success') {

         const songs = fetchData.songs || fetchData.output || [];

         if (songs.length > 0) {

           audio_url = songs[0].audio_url || songs[0].url;

           title = songs[0].title || 'Ma chanson';

           lyric = songs[0].lyric || songs[0].lyrics || '';

           break;

         }

       }


       // Progress animation

       const pct = Math.min(45 + (attempts / maxAttempts) * 45, 88);

       if (attempts < 6) setStep('Composition musicale…', pct);

       else if (attempts < 12) setStep('Enregistrement de la voix…', pct);

       else setStep('Mixage final…', pct);

     }


     if (!audio_url) throw new Error('Délai dépassé. Réessaie dans quelques instants.');


     setStep('Prête !', 100);

     await new Promise(r => setTimeout(r, 600));


     // Show result

     document.getElementById('loading').style.display = 'none';

     document.getElementById('result').style.display = 'block';

     document.getElementById('song-title').textContent = title;

     document.getElementById('song-meta').textContent = `${style.toUpperCase()} · ${mood} · ${lang}`;

     document.getElementById('audio-player').src = audio_url;

     document.getElementById('lyrics').textContent = lyric || '(Paroles non disponibles)';


   } catch (err) {

     document.getElementById('loading').style.display = 'none';

     const errEl = document.getElementById('error');

     errEl.style.display = 'block';

     errEl.textContent = '⚠️ ' + (err.message || 'Une erreur est survenue. Réessaie.');

     document.getElementById('btn-gen').disabled = false;

   }

 }


 function reset() {

   document.getElementById('result').style.display = 'none';

   document.getElementById('error').style.display = 'none';

   document.getElementById('btn-gen').disabled = false;

   document.getElementById('theme').value = '';

   document.getElementById('audio-player').src = '';

   window.scrollTo({ top: 0, behavior: 'smooth' });

 }
